export default { async fetch(request, env) { const headers = { "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Methods": "GET,OPTIONS", "Access-Control-Allow-Headers": "Content-Type", "Cache-Control": "no-store", }; try { if (request.method === "OPTIONS") return new Response(null, { headers }); const url = new URL(request.url); const studentId = (url.searchParams.get("studentId") || "").replace(/\D/g, ""); // مسار تحميل الملفات المباشر if (url.pathname.startsWith("/download/")) { const fileName = decodeURIComponent(url.pathname.replace("/download/", "")); const fileObj = await env.CERTS_BUCKET.get(fileName); if (!fileObj) return new Response("File Not Found", { status: 404, headers }); return new Response(fileObj.body, { headers: { ...headers, "Content-Type": "application/pdf", "Content-Disposition": `inline; filename="${fileName}"` } }); } // البحث عن الشهادات if (!studentId || studentId.length < 5) { return json({ success: false, error: "Invalid ID" }, 400, headers); } const databaseObj = await env.CERTS_BUCKET.get("database.txt"); if (!databaseObj) return json({ success: false, error: "Database not found" }, 500, headers); const content = await databaseObj.text(); const lines = content.split(/\r?\n/); const foundCerts = []; for (const line of lines) { if (!line.trim()) continue; const [id, courseName, fileName] = line.split(",").map(p => p.trim()); if (id === studentId) { foundCerts.push({ courseName, downloadUrl: `${url.origin}/download/${fileName}` }); } } return json({ success: true, count: foundCerts.length, certificates: foundCerts }, 200, headers); } catch (e) { return json({ success: false, error: e.message }, 500, headers); } } }; function json(obj, status, headers) { return new Response(JSON.stringify(obj), { status, headers: { ...headers, "Content-Type": "application/json" } }); }
